    ];

    const ROADSIGNS = [
        { id: 0, name: "STOP", description: "Come to a complete stop. Proceed only when safe. Mandatory control sign." },
        { id: 1, name: "GIVE WAY / YIELD", description: "Slow down, give priority to crossing traffic. Common in Botswana intersections." },

    // Botswana driving quiz (6 real questions)
    const QUIZ_QUESTIONS = [
        { question: "In Botswana, which side of the road should you drive on?", options: ["Left", "Right", "Center", "Any side"], correct: 0, explanation: "Botswana drives on the LEFT side." },
        { question: "What is the speed limit on a highway in Botswana?", options: ["80 km/h", "100 km/h", "120 km/h", "140 km/h"], correct: 2, explanation: "Highway speed limit is 120 km/h per Botswana road traffic act." },

        { question: "What does a solid white line on the road indicate?", options: ["Overtaking allowed", "No overtaking", "Parking only", "Slow down"], correct: 1, explanation: "Solid white line means no overtaking." },
        { question: "At a roundabout in Botswana, who has right of way?", options: ["Traffic entering", "Traffic already on roundabout", "Largest vehicle", "Pedestrians"], correct: 1, explanation: "Vehicles already on the roundabout have priority." }

    // --------------------------------------------------------------
    // GLOBAL STATE & localStorage
    // --------------------------------------------------------------
    let userData = {
        points: 0,
        completedLessons: [],   // ids
        masteredSigns: [],      // ids
        quizHighScore: 0,       // max points earned in quiz (out of 100)
        totalQuizAttempts: 0,
        theme: "default"        // default: black/gold/white
    };

    // Load from localStorage
    function loadStorage() {
        const saved = localStorage.getItem("eliteDrivingData");
        if(saved) {
            try {
                const parsed = JSON.parse(saved);
                userData = { ...userData, ...parsed };
            } catch(e) {}
        }
        // Recalc total points from achievements to stay consistent
        recalcTotalPoints();
        applyTheme(userData.theme);
    }

    function saveStorage() {
        localStorage.setItem("eliteDrivingData", JSON.stringify(userData));
    }

    // Recalculate points based on completed lessons, signs, quiz high score
    function recalcTotalPoints() {
        let total = 0;
        total += userData.completedLessons.length * 50;   // 50 per lesson
        total += userData.masteredSigns.length * 10;      // 10 per road sign
        total += userData.quizHighScore;                  // max 100 from quiz
        userData.points = total;
        saveStorage();
    }

    // complete lesson
    function completeLesson(lessonId) {
        if(!userData.completedLessons.includes(lessonId)) {
            userData.completedLessons.push(lessonId);
            recalcTotalPoints();
            saveStorage();
            showToast("✅ Lesson completed! +50 points", "gold");
            renderCurrentSection();
        } else {
            showToast("📘 Lesson already completed", "white");
        }
    }

    // master sign
    function masterSign(signId) {
        if(!userData.masteredSigns.includes(signId)) {
            userData.masteredSigns.push(signId);
            recalcTotalPoints();
            saveStorage();
            showToast("🛑 Road sign mastered! +10 points", "gold");
            renderCurrentSection();
        } else {
            showToast("🏆 Sign already learned", "white");
        }
    }

    // submit quiz: calculates score, updates highscore
    let currentQuizAnswers = new Array(QUIZ_QUESTIONS.length).fill(null);
    function renderQuizInterface() {
        let html = `<div class="glass-card" style="padding: 30px;"><h2 class="gold-text"><i class="fas fa-check-double"></i> Botswana Driving Quiz</h2>
        <p>Based on official handbook. Each correct answer = ~16.66 points (max 100). Highest score recorded.</p><div id="quizQuestionsContainer">`;
        QUIZ_QUESTIONS.forEach((q, idx) => {
            html += `<div class="lesson-item" style="margin-bottom: 20px;">
                        <h4>${idx+1}. ${q.question}</h4>
                        <div class="quiz-options-group" data-qidx="${idx}">`;
            q.options.forEach((opt, optIdx) => {
                const isSelected = (currentQuizAnswers[idx] === optIdx);
                html += `<div class="quiz-option ${isSelected ? 'selected' : ''}" data-qidx="${idx}" data-optidx="${optIdx}">${opt}</div>`;
            });
            html += `</div></div>`;
        });
        html += `<button id="submitQuizBtn" class="btn" style="margin-top: 20px;"><i class="fas fa-star"></i> Submit Quiz</button>
                <div id="quizFeedback" class="gold-text" style="margin-top: 20px;"></div></div>`;
        document.getElementById("quizQuestionsContainer").innerHTML = html;

        // attach listeners
        document.querySelectorAll('.quiz-option').forEach(opt => {
            opt.addEventListener('click', (e) => {
                const qIdx = parseInt(opt.dataset.qidx);
                const optIdx = parseInt(opt.dataset.optidx);
                currentQuizAnswers[qIdx] = optIdx;
                // re-render to show selected style (simple update)
                renderQuizInterface();
            });
        });
        document.getElementById("submitQuizBtn")?.addEventListener('click', () => {
            let score = 0;
            for(let i=0; i<QUIZ_QUESTIONS.length; i++) {
                if(currentQuizAnswers[i] === QUIZ_QUESTIONS[i].correct) score++;
            }
            const earnedPoints = Math.floor((score / QUIZ_QUESTIONS.length) * 100);
            let feedback = `You got ${score}/${QUIZ_QUESTIONS.length} correct. Points earned: ${earnedPoints}/100. `;
            let addedPoints = 0;
            if(earnedPoints > userData.quizHighScore) {
                addedPoints = earnedPoints - userData.quizHighScore;
                userData.quizHighScore = earnedPoints;
                recalcTotalPoints();
                saveStorage();
                feedback += `🎉 New highscore! +${addedPoints} points added.`;
            } else {
                feedback += `Your best remains ${userData.quizHighScore} points. Try again!`;
            }
            document.getElementById("quizFeedback").innerHTML = `<i class="fas fa-chart-simple"></i> ${feedback}`;
            renderCurrentSection(); // refresh progress/rewards
            showToast("Quiz submitted! Check progress.", "gold");
        });
    }

        toast.style.borderRadius = '30px';
        toast.style.zIndex = '9999';
        toast.style.backdropFilter = 'blur(10px)';
        toast.style.fontWeight = 'bold';
        document.body.appendChild(toast);
        setTimeout(() => toast.remove(), 2500);
    }

    function applyTheme(themeKey) {
        const theme = themes[themeKey] || themes.default;
        document.body.style.backgroundImage = theme.bgImage;
        document.body.style.backgroundSize = "cover";
        document.body.style.backgroundAttachment = "fixed";
        document.body.style.backgroundPosition = "center";
        document.documentElement.style.setProperty('--accent-gold', theme.primaryColor);
        document.documentElement.style.setProperty('--accent-glow', theme.primaryColor + "cc");
        const pseudo = document.querySelector('body::before');
        if(pseudo) {
            document.body.style.setProperty('--overlay-color', theme.bodyOverlay);
        }
        document.body.style.setProperty('--overlay-color', theme.bodyOverlay);
        // apply dynamic overlay effect
        let style = document.createElement('style');
        style.id = 'dynamic-theme-overlay';
        const existing = document.getElementById('dynamic-theme-overlay');
        if(existing) existing.remove();
        style.id = 'dynamic-theme-overlay';
        style.innerText = `body::before { background: ${theme.bodyOverlay} !important; backdrop-filter: blur(3px); }`;
        document.head.appendChild(style);
    }

        const badges = [

            { name: "Silver Driver", need: 250, icon: "fa-star", unlocked: userData.points >= 250 },
            { name: "Gold Roadmaster", need: 500, icon: "fa-crown", unlocked: userData.points >= 500 },
            { name: "Elite Instructor", need: 1000, icon: "fa-gem", unlocked: userData.points >= 1000 }
        ];
        let html = `<h2 class="gold-text"><i class="fas fa-gift"></i> Achievements & Rewards</h2><div class="grid-cards">`;
        badges.forEach(b => {
            html += `<div class="glass-card stat-card ${b.unlocked ? 'reward-badge unlocked' : ''}" style="background:${b.unlocked ? 'rgba(255,215,0,0.2)' : ''}">
                        <i class="fas ${b.icon} fa-2x gold-text"></i><h3>${b.name}</h3>
                        <p>Requires ${b.need} XP</p>${b.unlocked ? '<span class="gold-text">✔️ UNLOCKED</span>' : `<span>🔒 Locked</span>`}
                    </div>`;
        });
        html += `</div><div class="glass-card" style="margin-top:20px; padding:20px;"><i class="fas fa-gold"></i> Keep driving & quizzing to earn more XP!</div>`;
        container.innerHTML = html;
    }

    function renderSettings(container) {
        let themeHtml = `<h2 class="gold-text"><i class="fas fa-palette"></i> Visual Settings</h2>
        <div class="glass-card" style="padding: 25px;">
            <label>Select Theme: </label><select id="themeSelect">`;
        for(let [key, val] of Object.entries(themes)) {
            themeHtml += `<option value="${key}" ${userData.theme === key ? 'selected' : ''}>${val.name}</option>`;
        }
        themeHtml += `</select><br><br><button id="applyThemeBtn" class="btn">Apply Theme</button><hr><h3>Reset Progress?</h3><button id="resetProgressBtn" class="btn" style="background:darkred;">Reset All Data</button></div>`;
        container.innerHTML = themeHtml;
        document.getElementById("themeSelect").addEventListener("change", (e) => {
            userData.theme = e.target.value;
            applyTheme(userData.theme);
            saveStorage();
            showToast(`Theme changed to ${themes[userData.theme].name}`, "gold");
        });
        document.getElementById("applyThemeBtn").addEventListener("click", () => {
            const sel = document.getElementById("themeSelect").value;
            userData.theme = sel;
            applyTheme(sel);
            saveStorage();
            showToast("Theme applied!", "gold");
        });
        document.getElementById("resetProgressBtn").addEventListener("click", () => {
            if(confirm("⚠️ WARNING: This will reset all lessons, signs, quiz scores and points. Continue?")) {
                userData = {
                    points: 0,
                    completedLessons: [],
                    masteredSigns: [],
                    quizHighScore: 0,
                    totalQuizAttempts: 0,
                    theme: userData.theme
                };
                recalcTotalPoints();
                saveStorage();
                currentQuizAnswers.fill(null);
                renderCurrentSection();
                showToast("All progress has been reset!", "gold");
            }
        });
    }

    // navigation + init
    function initNavigation() {
        document.querySelectorAll('.nav-btn').forEach(btn => {
            btn.addEventListener('click', (e) => {
                document.querySelectorAll('.nav-btn').forEach(b => b.classList.remove('active'));
                btn.classList.add('active');
                activeSection = btn.dataset.section;
                renderCurrentSection();
            });
        });
    }

    loadStorage();
    initNavigation();
    renderCurrentSection();
</script>
